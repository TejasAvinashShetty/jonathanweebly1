f(0)  &= e^{-\lambda A} B e^{\lambda A} \big\vert_{\lambda=0} \\
      &= e^{-(0) A} B e^{(0) A} \big\vert_{\lambda=0} \\
      &= I B I \\
      &= B \\
\\
f'(0) &= e^{-\lambda A} [ B,A ] e^{\lambda A} \big\vert_{\lambda=0} \\
      &= e^{-(0) A} [B,A] e^{(0) A} \\
      &=I [B,A] I \\
      &= [B,A]
\intertext{Similarly,}            
f''(0) &= e^{-\lambda A} \left[ [ B,A ], A \right] e^{\lambda A} \big\vert_{\lambda=0} \\
       &= e^{-(0) A} [[B,A],A] e^{(0) A} \\
      &=I [[B,A],A] I \\
      &= [[B,A],A]
\end{align*}
